Class: RemoteViewController

X.util. RemoteViewController

X.util.RemoteViewController 클래스는 원격지에 있는 화면을 불러와 화면을 전환하도록 한다.

new RemoteViewController()

Properties:
Name Type Description
config.initPage String 초기 화면의 url 을 지정한다.
config.transition String 화면 전환 animation 종류를 지정한다. slide
slidefade
slideup
slidedown
pop
fade
flip
turn
flow

Example
var view = new X.View({
     viewController: new X.util.RemoteViewController({
         beforenextchange: function(){ },        //다음 화면으로 전환하기 직전에 호출된다.
         afternextchange: function(){ },         //다음 화면으로 전환한 후에 호출된다.
         beforeprevchange: function(){ },        //이전 화면으로 전환하기 진적에 호출된다.
         afterprevchange: function(){ }          //이전 화면으로 전환한 후에 호출된다.
     }),
     children: [
         new X.View(),
         new X.View(),
         new X.View()
     ]
});

var vc = view.getViewController();
vc.initPage({ url: "page.html" });
vc.nextPage({ url: "page-next.html" });
