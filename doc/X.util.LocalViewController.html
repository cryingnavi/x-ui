Class: LocalViewController

X.util. LocalViewController

new LocalViewController()

X.util.LocalViewController 클래스는 생성된 X.View 의 자식 view 들에 대해서 화면전환을 제공한다.
Properties:
Name Type Description
config.activeIndex String 처음 생성시에 활성화 될 화면의 인덱스
config.transition String 화면 전환 animation 종류를 지정한다.
slide, slidefade, slideup, slidedown
pop, fade, flip, turn, flow
roll
Source:
  • x-ui.js
Example
var view = new X.View({
     viewController: new X.util.LocalViewController({
         beforenextchange: function(){ },        //다음 화면으로 전환하기 직전에 호출된다.
         afternextchange: function(){ },         //다음 화면으로 전환한 후에 호출된다.
         beforeprevchange: function(){ },        //이전 화면으로 전환하기 진적에 호출된다.
         afterprevchange: function(){ }          //이전 화면으로 전환한 후에 호출된다.
     }),
     children: [
         new X.View(),
         new X.View(),
         new X.View()
     ]
});

var vc = view.getViewController();
vc.nextPage(1);
